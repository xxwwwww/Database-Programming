
CREATE TABLE PLAYER(
PLAYER_ID     SERIAL,
TEAM_ID          INT     NOT NULL,
UNIFORM_NUM       INT         NOT NULL,
FIRST_NAME    VARCHAR(30)      NOT NULL,
LAST_NAME       VARCHAR(30)         NOT NULL,
MPG         INT                                 NOT NULL,
PPG         INT            NOT NULL,
RPG         INT            NOT NULL,
APG         INT           NOT NULL,
SPG        FLOAT       NOT NULL,
BPG        FLOAT        NOT NULL,
    CONSTRAINT PLAYER_ID PRIMARY KEY (PLAYER_ID)
);

CREATE TABLE TEAM(
TEAM_ID 		SERIAL,
NAME 	VARCHAR(50)		NOT NULL,
STATE_ID 		INT 		NOT NULL,
COLOR_ID		INT		NOT NULL,
WINS			INT  		NOT NULL,
LOSSES 		INT		NOT NULL,
    CONSTRAINT TEAM_ID PRIMARY KEY (TEAM_ID)
);



CREATE TABLE STATE(
STATE_ID 		SERIAL,
NAME		VARCHAR		NOT NULL,
    CONSTRAINT STATE_ID PRIMARY KEY (STATE_ID)
);



CREATE TABLE COLOR(
COLOR_ID 		SERIAL,
NAME		VARCHAR(50)		NOT NULL,
    CONSTRAINT COLOR_ID PRIMARY KEY (COLOR_ID)
);
